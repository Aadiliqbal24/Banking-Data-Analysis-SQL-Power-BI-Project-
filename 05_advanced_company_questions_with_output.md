## Q2. Anomaly check: transactions > 3x account's own average

```sql
WITH account_avg AS (
    SELECT account_id, AVG(amount) AS avg_amount
    FROM Fact_Transaction GROUP BY account_id
)
SELECT f.transaction_id, f.account_id, f.amount, ROUND(a.avg_amount,2) AS avg_amount,
       ROUND(f.amount / a.avg_amount, 2) AS times_above_avg
FROM Fact_Transaction f
JOIN account_avg a ON a.account_id = f.account_id
WHERE f.amount > 3 * a.avg_amount
ORDER BY times_above_avg DESC;
```

## Q6. H1 vs H2 growth % per branch

```sql
WITH half_year_totals AS (
    SELECT b.branch_name,
           CASE WHEN d.month <= 6 THEN 'H1' ELSE 'H2' END AS half_year,
           SUM(f.amount) AS total_amount
    FROM Fact_Transaction f
    JOIN Dim_Branch b ON b.branch_id = f.branch_id
    JOIN Dim_Date d   ON d.date_id = f.date_id
    GROUP BY b.branch_name, CASE WHEN d.month <= 6 THEN 'H1' ELSE 'H2' END
)
SELECT branch_name,
       SUM(CASE WHEN half_year = 'H1' THEN total_amount ELSE 0 END) AS h1_total,
       SUM(CASE WHEN half_year = 'H2' THEN total_amount ELSE 0 END) AS h2_total,
       ROUND(
         (SUM(CASE WHEN half_year = 'H2' THEN total_amount ELSE 0 END)
        - SUM(CASE WHEN half_year = 'H1' THEN total_amount ELSE 0 END))
        / NULLIF(SUM(CASE WHEN half_year = 'H1' THEN total_amount ELSE 0 END), 0) * 100, 2
       ) AS h1_to_h2_growth_pct
FROM half_year_totals
GROUP BY branch_name
ORDER BY h1_to_h2_growth_pct DESC;
```

## Q7. Dormant-but-valuable accounts (reactivation list)

```sql
WITH company_avg AS (SELECT AVG(amount) AS overall_avg FROM Fact_Transaction),
account_avg AS (SELECT account_id, AVG(amount) AS acct_avg FROM Fact_Transaction GROUP BY account_id)
SELECT a.account_id, c.customer_name, ROUND(aa.acct_avg,2) AS acct_avg, ROUND(ca.overall_avg,2) AS overall_avg
FROM Dim_Account a
JOIN Dim_Customer c ON c.customer_id = a.customer_id
JOIN account_avg aa ON aa.account_id = a.account_id
CROSS JOIN company_avg ca
WHERE a.status = 'Inactive' AND aa.acct_avg > ca.overall_avg
ORDER BY aa.acct_avg DESC;
```

## Q10. Composite branch score (weighted, normalized)

```sql
WITH branch_stats AS (
    SELECT b.branch_id, b.branch_name, COUNT(*) AS txn_count, SUM(f.amount) AS txn_value
    FROM Fact_Transaction f JOIN Dim_Branch b ON b.branch_id = f.branch_id
    GROUP BY b.branch_id, b.branch_name
),
normalized AS (
    SELECT branch_name, txn_count, txn_value,
           (txn_count - MIN(txn_count) OVER ()) * 100.0
             / NULLIF(MAX(txn_count) OVER () - MIN(txn_count) OVER (), 0) AS count_score,
           (txn_value - MIN(txn_value) OVER ()) * 100.0
             / NULLIF(MAX(txn_value) OVER () - MIN(txn_value) OVER (), 0) AS value_score
    FROM branch_stats
)
SELECT branch_name, ROUND(count_score * 0.4 + value_score * 0.6, 2) AS composite_score
FROM normalized
ORDER BY composite_score DESC;
```

| --- | --- |

| Mumbai Branch | 91.14 |
| Kochi Branch | 71.93 |
| Madurai Branch | 50.56 |

## Q11. Same-day, same-amount, same-branch pairs (collusion check)

```sql

       c1.customer_name AS customer_1, c2.customer_name AS customer_2

     ON f1.branch_id = f2.branch_id
```

## Q12. 7-day moving average of daily transaction value

```sql
WITH daily_totals AS (
    SELECT d.full_date, SUM(f.amount) AS daily_amount
    FROM Fact_Transaction f JOIN Dim_Date d ON d.date_id = f.date_id
    GROUP BY d.full_date
)
SELECT full_date, daily_amount,
       ROUND(AVG(daily_amount) OVER (
           ORDER BY full_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
       ), 2) AS moving_avg_7day
FROM daily_totals
ORDER BY full_date;
```

## Q14. 2x2 customer segmentation (income x activity)

```sql
WITH customer_metrics AS (
    SELECT c.customer_id, c.customer_name, c.annual_income,
           COALESCE(SUM(f.amount), 0) AS total_txn_value
    FROM Dim_Customer c
    JOIN Dim_Account a ON a.customer_id = c.customer_id
    LEFT JOIN Fact_Transaction f ON f.account_id = a.account_id
    GROUP BY c.customer_id, c.customer_name, c.annual_income
),
scored AS (
    SELECT *,
           NTILE(2) OVER (ORDER BY annual_income DESC)   AS income_half,
           NTILE(2) OVER (ORDER BY total_txn_value DESC) AS activity_half
    FROM customer_metrics
)
SELECT customer_id, customer_name, annual_income, total_txn_value,
       CASE
           WHEN income_half = 1 AND activity_half = 1 THEN 'High Value / High Activity'
           WHEN income_half = 1 AND activity_half = 2 THEN 'High Value / Low Activity'
           WHEN income_half = 2 AND activity_half = 1 THEN 'Low Value / High Activity'
           ELSE 'Low Value / Low Activity'
       END AS customer_segment
FROM scored
ORDER BY annual_income DESC;
```
